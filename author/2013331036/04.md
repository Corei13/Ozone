string s;
int Find (int v, int q, int n) {
    int i, j, k, l;
    for (i = v; (i + q + n) % n != v; i = (i + q + n) % n) {
        if (s[i] != 'w') {
            break;
        }
    }

    return s[i];
}

main() {
    freopen ("beads.in", "r", stdin);
    freopen ("beads.out", "w", stdout);
    int i, j, k, l;
    int n;
    cin >> n >> s;
    int ans = 0;
    for (int z = 0; z < n; z++) {
        int sum = 0;
        k = Find (z, 1, n);
        for (i = z;; i = l) {
            sum++;
            l = (i + 1) % n;
            if (s[l] != 'w' && s[l] != k) {
                break;
            }
            if (l == z) {
                break;
            }
        }

        k = Find ((z - 1 + n) % n, -1, n);
        for (j = (z - 1 + n) % n; j != i; j = (j - 1 + n) % n) {
            if (s[j] != 'w' && s[j] != k) {
                break;
            }
            sum++;
            if ((j - 1 + n) % n == i) {
                break;
            }
        }

        ans = max (ans, sum);

    }

    cout << ans << '\n';
    return 0;

}
